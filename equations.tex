\section{Hydraulic variables}
\[\begin{aligned}
& H = S-B\\
& S = H+B\\
& B = S-H\\
& M = \sqrt{U^2+V^2}\\
& I = H\times U\\
& J = H\times V\\
& Q = \sqrt{I^2 + J^2}\\
& C = \sqrt{g\,H}\\
& F = \frac{M}{C}
\end{aligned}\]
